# CDC with AWS DMS Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a CDC pipeline using AWS DMS that captures changes from RDS PostgreSQL into S3, reconstructs current state with a Glue job, and characterizes DMS behavior (lag, delivery semantics, event format, reconstruction correctness) under realistic and stress workloads.

**Architecture:** DMS runs in Full Load + CDC mode against RDS PostgreSQL, writing CSV change events to S3 (`change-log/`). A scheduled Glue job reads new DMS files, applies op codes (I/U/D) to a current-state Parquet dataset in S3, and advances a watermark JSON file. DuckDB validates reconstruction correctness locally by diffing Parquet against live RDS.

**Tech Stack:** Python 3.12, AWS DMS, RDS PostgreSQL, AWS Glue (Python shell), S3 (CSV + Parquet), DuckDB, boto3, psycopg2, pandas, matplotlib

-- Single OLTP table for CDC mechanics study
CREATE TABLE IF NOT EXISTS transactions (
    id          SERIAL        PRIMARY KEY,
    user_id     INTEGER       NOT NULL,
    amount      NUMERIC(10,2) NOT NULL,
    status      VARCHAR(20)   NOT NULL DEFAULT 'pending',
    metadata    JSONB,
    version     INTEGER       NOT NULL DEFAULT 1,
    created_at  TIMESTAMPTZ   NOT NULL DEFAULT NOW(),
    updated_at  TIMESTAMPTZ   NOT NULL DEFAULT NOW()
);

-- Auto-increment version and refresh updated_at on every UPDATE
CREATE OR REPLACE FUNCTION trg_transactions_stamp()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = NOW();
    NEW.version    = OLD.version + 1;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER transactions_stamp
    BEFORE UPDATE ON transactions
    FOR EACH ROW EXECUTE FUNCTION trg_transactions_stamp();

-- Required for DMS to capture full row on DELETE (not just PK)
ALTER TABLE transactions REPLICA IDENTITY FULL;
```

"""
Seed the RDS transactions table with initial rows before starting DMS.
Run once: python seed_table.py

Env vars required:
    RDS_HOST, RDS_PORT, RDS_DB, RDS_USER, RDS_PASSWORD
"""
import os
import json
import random
import psycopg2
from faker import Faker

fake = Faker()
Faker.seed(42)
random.seed(42)

RDS_HOST     = os.environ["RDS_HOST"]
RDS_PORT     = int(os.environ.get("RDS_PORT", 5432))
RDS_DB       = os.environ["RDS_DB"]
RDS_USER     = os.environ["RDS_USER"]
RDS_PASSWORD = os.environ["RDS_PASSWORD"]

TOTAL_ROWS = 10_000
BATCH_SIZE = 500
STATUSES   = ["pending", "processing", "completed", "cancelled"]

def make_row():
    return (
        random.randint(1, 2000),
        round(random.uniform(5.0, 9999.99), 2),
        random.choice(STATUSES),
        json.dumps({"source": fake.word(), "region": fake.country_code()}),
    )

def main():
    conn = psycopg2.connect(
        host=RDS_HOST, port=RDS_PORT,
        dbname=RDS_DB, user=RDS_USER, password=RDS_PASSWORD,
    )
    conn.autocommit = False

    with conn.cursor() as cur:
        for batch_start in range(0, TOTAL_ROWS, BATCH_SIZE):
            rows = [make_row() for _ in range(BATCH_SIZE)]
            cur.executemany(
                "INSERT INTO transactions (user_id, amount, status, metadata) VALUES (%s, %s, %s, %s)",
                rows,
            )
            conn.commit()
            print(f"Inserted rows {batch_start + BATCH_SIZE}/{TOTAL_ROWS}")

    conn.close()
    print("Seeding complete.")

if __name__ == "__main__":
    main()
```

```python
"""
One-time audit of DMS S3 output.
Reads sample LOAD and CDC files, prints schema, op code distribution,
column completeness, and type fidelity.

Usage: python inspect_dms_output.py --bucket <bucket> --prefix change-log/
"""
import argparse
import io
import csv
from collections import Counter
import boto3

def list_keys(s3, bucket, prefix):
    paginator = s3.get_paginator("list_objects_v2")
    keys = []
    for page in paginator.paginate(Bucket=bucket, Prefix=prefix):
        for obj in page.get("Contents", []):
            keys.append(obj["Key"])
    return sorted(keys)

def read_csv_from_s3(s3, bucket, key):
    obj = s3.get_object(Bucket=bucket, Key=key)
    content = obj["Body"].read().decode("utf-8")
    reader = csv.DictReader(io.StringIO(content))
    rows = list(reader)
    return rows, reader.fieldnames

def audit_prefix(s3, bucket, prefix, sample_size=2):
    keys = list_keys(s3, bucket, prefix)
    print(f"\n{'='*60}")
    print(f"Prefix : {prefix}")
    print(f"Files  : {len(keys)}")
    if not keys:
        print("  (no files found)")
        return

    all_rows, columns = [], None
    for key in keys[:sample_size]:
        rows, fieldnames = read_csv_from_s3(s3, bucket, key)
        all_rows.extend(rows)
        columns = fieldnames
        print(f"\nFile   : {key}  ({len(rows)} rows)")
        print(f"Columns: {fieldnames}")
        if rows:
            print(f"Sample : {rows[0]}")

    if columns and "Op" in columns:
        op_counts = Counter(r.get("Op", "?") for r in all_rows)
        print(f"\nOp distribution: {dict(op_counts)}")

    if columns and all_rows:
        print("\nColumn completeness (% non-empty):")
        for col in columns:
            pct = 100 * sum(1 for r in all_rows if r.get(col, "").strip()) / len(all_rows)
            print(f"  {col:20s}: {pct:.1f}%")

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--bucket", required=True)
    parser.add_argument("--prefix", default="change-log/")
    args = parser.parse_args()

    s3 = boto3.client("s3")
    audit_prefix(s3, args.bucket, args.prefix + "LOAD/")
    audit_prefix(s3, args.bucket, args.prefix + "CDC/")

if __name__ == "__main__":
    main()
```

```python
"""
Validates that current-state Parquet in S3 matches live RDS transactions table.
Run after any Glue reconstruction job run.

Usage:
    python validate_reconstruction.py --bucket <bucket>

Env vars required:
    RDS_HOST, RDS_PORT, RDS_DB, RDS_USER, RDS_PASSWORD
"""
import os
import argparse
import io
import boto3
import duckdb
import psycopg2

RDS_HOST     = os.environ["RDS_HOST"]
RDS_PORT     = int(os.environ.get("RDS_PORT", 5432))
RDS_DB       = os.environ["RDS_DB"]
RDS_USER     = os.environ["RDS_USER"]
RDS_PASSWORD = os.environ["RDS_PASSWORD"]

def fetch_rds_rows(conn):
    with conn.cursor() as cur:
        cur.execute("SELECT id, user_id, CAST(amount AS TEXT), status, version FROM transactions ORDER BY id")
        return {row[0]: row for row in cur.fetchall()}

def fetch_parquet_rows(s3, bucket, prefix):
    paginator = s3.get_paginator("list_objects_v2")
    keys = []
    for page in paginator.paginate(Bucket=bucket, Prefix=prefix):
        for obj in page.get("Contents", []):
            if obj["Key"].endswith(".parquet"):
                keys.append(obj["Key"])
    if not keys:
        return {}

    duck = duckdb.connect()
    frames = {}
    for key in keys:
        obj = s3.get_object(Bucket=bucket, Key=key)
        data = obj["Body"].read()
        rows = duck.execute(
            "SELECT id, user_id, CAST(amount AS VARCHAR), status, version FROM read_parquet(?)",
            [io.BytesIO(data)]
        ).fetchall()
        for row in rows:
            frames[row[0]] = row
    duck.close()
    return frames

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--bucket", required=True)
    parser.add_argument("--prefix", default="current-state/transactions/")
    args = parser.parse_args()

    conn = psycopg2.connect(
        host=RDS_HOST, port=RDS_PORT,
        dbname=RDS_DB, user=RDS_USER, password=RDS_PASSWORD,
    )
    s3 = boto3.client("s3")

    rds = fetch_rds_rows(conn)
    pq  = fetch_parquet_rows(s3, args.bucket, args.prefix)
    conn.close()

    only_rds = set(rds) - set(pq)
    only_pq  = set(pq)  - set(rds)
    both     = set(rds) & set(pq)
    mismatch = [i for i in both if rds[i] != pq[i]]

    total    = len(rds)
    correct  = total - len(only_rds) - len(only_pq) - len(mismatch)
    accuracy = 100 * correct / max(total, 1)

    print(f"\nRDS rows     : {total}")
    print(f"Parquet rows : {len(pq)}")
    print(f"Only in RDS  : {len(only_rds)}")
    print(f"Only in S3   : {len(only_pq)}")
    print(f"Mismatches   : {len(mismatch)}")
    print(f"Accuracy     : {accuracy:.2f}%")

    if only_rds:  print(f"\nSample IDs only in RDS : {sorted(only_rds)[:5]}")
    if only_pq:   print(f"Sample IDs only in S3  : {sorted(only_pq)[:5]}")
    if mismatch:  print(f"Sample mismatches      : {[(i, rds[i], pq[i]) for i in mismatch[:3]]}")

if __name__ == "__main__":
    main()
```

- [ ] **Step 3: Write reconstruct.py — full file including seed mode**

```python
"""
Glue Python shell job — runs every 5 minutes.
Reads new DMS change-log files from S3, applies op codes (I/U/D) to
current-state Parquet, and advances the watermark.

Glue job parameters:
    --S3_BUCKET            S3 bucket name
    --CHANGE_LOG_PREFIX    e.g. change-log
    --CURRENT_STATE_PREFIX e.g. current-state/transactions
    --WATERMARK_KEY        e.g. watermark/reconstruct.json
"""
import sys
import json
import io
import boto3
import pandas as pd
from awsglue.utils import getResolvedOptions

args = getResolvedOptions(sys.argv, [
    "S3_BUCKET", "CHANGE_LOG_PREFIX", "CURRENT_STATE_PREFIX", "WATERMARK_KEY",
])
BUCKET         = args["S3_BUCKET"]
CHANGE_LOG_PFX = args["CHANGE_LOG_PREFIX"]
CURRENT_STATE  = args["CURRENT_STATE_PREFIX"]
WATERMARK_KEY  = args["WATERMARK_KEY"]

s3 = boto3.client("s3")

# ---------- helpers ----------

def read_watermark():
    try:
        obj = s3.get_object(Bucket=BUCKET, Key=WATERMARK_KEY)
        return json.loads(obj["Body"].read())
    except s3.exceptions.NoSuchKey:
        return {"seeded": False, "processed_keys": []}

def write_watermark(wm):
    s3.put_object(Bucket=BUCKET, Key=WATERMARK_KEY, Body=json.dumps(wm).encode())

def list_keys(prefix):
    paginator = s3.get_paginator("list_objects_v2")
    keys = []
    for page in paginator.paginate(Bucket=BUCKET, Prefix=prefix):
        for obj in page.get("Contents", []):
            keys.append(obj["Key"])
    return sorted(keys)

def read_csv_s3(key):
    obj = s3.get_object(Bucket=BUCKET, Key=key)
    return pd.read_csv(io.BytesIO(obj["Body"].read()), dtype=str)

def read_current_state():
    key = f"{CURRENT_STATE}/data.parquet"
    try:
        obj = s3.get_object(Bucket=BUCKET, Key=key)
        return pd.read_parquet(io.BytesIO(obj["Body"].read()))
    except Exception:
        return pd.DataFrame()

def write_current_state(df):
    key = f"{CURRENT_STATE}/data.parquet"
    buf = io.BytesIO()
    df.to_parquet(buf, index=False)
    buf.seek(0)
    s3.put_object(Bucket=BUCKET, Key=key, Body=buf.getvalue())

# ---------- seed mode ----------

def seed_from_load_files():
    """Read all LOAD files and write initial current-state Parquet."""
    load_keys = list_keys(f"{CHANGE_LOG_PFX}/LOAD/")
    if not load_keys:
        raise RuntimeError("No LOAD files found — run DMS full load first.")

    frames = []
    for key in load_keys:
        df = read_csv_s3(key)
        frames.append(df)
        print(f"  Read LOAD file: {key} ({len(df)} rows)")

    full_df = pd.concat(frames, ignore_index=True)
    write_current_state(full_df)
    print(f"Seed complete. Wrote {len(full_df)} rows to current-state.")

# ---------- CDC merge ----------

def apply_cdc_events(wm):
    """Read new CDC files and apply ops to current-state Parquet."""
    all_cdc_keys = list_keys(f"{CHANGE_LOG_PFX}/CDC/")
    processed    = set(wm.get("processed_keys", []))
    new_keys     = [k for k in all_cdc_keys if k not in processed]

    if not new_keys:
        print("No new CDC files to process.")
        return wm

    frames = [read_csv_s3(k) for k in sorted(new_keys)]
    events = pd.concat(frames, ignore_index=True)

    # Enforce ordering; de-duplicate
    events = events.sort_values(["_timestamp", "id"], ascending=True)
    events = events.drop_duplicates(subset=["id", "_timestamp", "Op"], keep="last")

    current_df = read_current_state()

    for _, row in events.iterrows():
        op     = row.get("Op", "").strip().upper()
        row_id = str(row["id"])

        if op == "I":
            new_row = row.drop("Op").to_frame().T
            current_df = pd.concat([current_df, new_row], ignore_index=True)

        elif op == "U":
            current_df = current_df[current_df["id"].astype(str) != row_id]
            new_row = row.drop("Op").to_frame().T
            current_df = pd.concat([current_df, new_row], ignore_index=True)

        elif op == "D":
            current_df = current_df[current_df["id"].astype(str) != row_id]

    write_current_state(current_df)
    wm["processed_keys"] = sorted(processed | set(new_keys))
    print(f"Applied {len(events)} events. Current state: {len(current_df)} rows.")
    return wm

# ---------- main ----------

def main():
    wm = read_watermark()

    if not wm["seeded"]:
        print("Seed mode: processing LOAD files...")
        seed_from_load_files()
        wm["seeded"] = True
        wm["processed_keys"] = []
        write_watermark(wm)
        return

    wm = apply_cdc_events(wm)
    write_watermark(wm)

main()
```

```python
"""
Glue Python shell job — runs every 5 minutes.
Generates a realistic batch of OLTP changes on the transactions table.
Records committed_at timestamps to S3 for lag analysis.

Glue job parameters:
    --RDS_HOST, --RDS_PORT, --RDS_DB, --RDS_USER, --RDS_PASSWORD
    --S3_BUCKET, --METRICS_PREFIX
"""
import sys
import json
import random
import io
import csv
import boto3
import psycopg2
from datetime import datetime, timezone
from awsglue.utils import getResolvedOptions

args = getResolvedOptions(sys.argv, [
    "RDS_HOST", "RDS_PORT", "RDS_DB", "RDS_USER", "RDS_PASSWORD",
    "S3_BUCKET", "METRICS_PREFIX",
])

RDS_HOST     = args["RDS_HOST"]
RDS_PORT     = int(args["RDS_PORT"])
RDS_DB       = args["RDS_DB"]
RDS_USER     = args["RDS_USER"]
RDS_PASSWORD = args["RDS_PASSWORD"]
BUCKET       = args["S3_BUCKET"]
METRICS_PFX  = args["METRICS_PREFIX"]

TRANSITIONS  = {"pending": "processing", "processing": "completed", "completed": "cancelled"}
INSERT_COUNT = random.randint(50, 100)
UPDATE_COUNT = random.randint(30, 50)
DELETE_COUNT = random.randint(5, 10)

s3 = boto3.client("s3")

def now_utc():
    return datetime.now(timezone.utc).isoformat()

def run_batch(conn):
    events = []

    with conn.cursor() as cur:
        # INSERTs
        for _ in range(INSERT_COUNT):
            cur.execute("""
                INSERT INTO transactions (user_id, amount, status, metadata)
                VALUES (%s, %s, 'pending', %s) RETURNING id
            """, (
                random.randint(1, 2000),
                round(random.uniform(5.0, 9999.99), 2),
                json.dumps({"source": random.choice(["web", "mobile", "api"])}),
            ))
            row_id = cur.fetchone()[0]
            conn.commit()
            events.append({"op": "I", "id": row_id, "committed_at": now_utc()})

        # UPDATEs — status transitions on existing rows
        cur.execute("""
            SELECT id, status FROM transactions
            WHERE status IN ('pending', 'processing')
            ORDER BY RANDOM() LIMIT %s
        """, (UPDATE_COUNT,))
        for row_id, status in cur.fetchall():
            new_status = TRANSITIONS.get(status, "completed")
            cur.execute("UPDATE transactions SET status = %s WHERE id = %s", (new_status, row_id))
            conn.commit()
            events.append({"op": "U", "id": row_id, "committed_at": now_utc()})

        # DELETEs — remove cancelled rows
        cur.execute("""
            SELECT id FROM transactions
            WHERE status = 'cancelled'
            ORDER BY RANDOM() LIMIT %s
        """, (DELETE_COUNT,))
        for (row_id,) in cur.fetchall():
            cur.execute("DELETE FROM transactions WHERE id = %s", (row_id,))
            conn.commit()
            events.append({"op": "D", "id": row_id, "committed_at": now_utc()})

    return events

def write_metrics(events):
    ts  = datetime.now(timezone.utc).strftime("%Y%m%dT%H%M%SZ")
    key = f"{METRICS_PFX}/{ts}.csv"
    buf = io.StringIO()
    writer = csv.DictWriter(buf, fieldnames=["op", "id", "committed_at"])
    writer.writeheader()
    writer.writerows(events)
    s3.put_object(Bucket=BUCKET, Key=key, Body=buf.getvalue().encode())
    print(f"Wrote {len(events)} events to s3://{BUCKET}/{key}")

def main():
    conn = psycopg2.connect(
        host=RDS_HOST, port=RDS_PORT,
        dbname=RDS_DB, user=RDS_USER, password=RDS_PASSWORD,
    )
    events = run_batch(conn)
    conn.close()
    write_metrics(events)
    i = sum(1 for e in events if e["op"] == "I")
    u = sum(1 for e in events if e["op"] == "U")
    d = sum(1 for e in events if e["op"] == "D")
    print(f"Done. I={i} U={u} D={d}")

main()
```

## Task 7: Stress workload

**Files:**
- Modify: `2-cdc-dms/data/simulate_stress.py`

- [ ] **Step 1: Write simulate_stress.py**

```python
"""
Stress experiments targeting DMS edge cases.
Run manually, one scenario at a time, with the reconstruction job running.

Usage:
    python simulate_stress.py --scenario burst      --output analysis/stress_burst.csv
    python simulate_stress.py --scenario concurrent --output analysis/stress_concurrent.csv
    python simulate_stress.py --scenario ghost      --output analysis/stress_ghost.csv
    python simulate_stress.py --scenario large-txn  --output analysis/stress_large_txn.csv
    python simulate_stress.py --scenario rollback   --output analysis/stress_rollback.csv

Env vars required:
    RDS_HOST, RDS_PORT, RDS_DB, RDS_USER, RDS_PASSWORD
"""
import os
import argparse
import json
import random
import threading
import csv
from datetime import datetime, timezone
import psycopg2

RDS_HOST     = os.environ["RDS_HOST"]
RDS_PORT     = int(os.environ.get("RDS_PORT", 5432))
RDS_DB       = os.environ["RDS_DB"]
RDS_USER     = os.environ["RDS_USER"]
RDS_PASSWORD = os.environ["RDS_PASSWORD"]

def connect():
    return psycopg2.connect(
        host=RDS_HOST, port=RDS_PORT,
        dbname=RDS_DB, user=RDS_USER, password=RDS_PASSWORD,
    )

def now_utc():
    return datetime.now(timezone.utc).isoformat()

def log_event(events, op, row_id, note=""):
    events.append({"op": op, "id": row_id, "triggered_at": now_utc(), "note": note})

# Scenario 1: Burst — 50 rapid UPDATEs to the same row
def scenario_burst(events):
    conn = connect()
    conn.autocommit = True
    with conn.cursor() as cur:
        cur.execute(
            "INSERT INTO transactions (user_id, amount, status) VALUES (1, 1.00, 'pending') RETURNING id"
        )
        row_id = cur.fetchone()[0]
        log_event(events, "I", row_id, "burst-seed")
        for i in range(50):
            cur.execute("UPDATE transactions SET amount = %s WHERE id = %s", (float(i), row_id))
            log_event(events, "U", row_id, f"burst-{i}")
    conn.close()
    print(f"Burst: id={row_id}, 50 UPDATEs fired")

# Scenario 2: Concurrent — 5 threads updating the same row simultaneously
def scenario_concurrent(events):
    conn = connect()
    conn.autocommit = True
    with conn.cursor() as cur:
        cur.execute(
            "INSERT INTO transactions (user_id, amount, status) VALUES (2, 100.00, 'pending') RETURNING id"
        )
        row_id = cur.fetchone()[0]
        log_event(events, "I", row_id, "concurrent-seed")
    conn.close()

    def worker(worker_id):
        c = connect()
        c.autocommit = True
        with c.cursor() as cur:
            cur.execute(
                "UPDATE transactions SET amount = %s, metadata = %s WHERE id = %s",
                (float(worker_id * 10), json.dumps({"worker": worker_id}), row_id),
            )
        log_event(events, "U", row_id, f"concurrent-worker-{worker_id}")
        c.close()

    threads = [threading.Thread(target=worker, args=(i,)) for i in range(5)]
    for t in threads: t.start()
    for t in threads: t.join()
    print(f"Concurrent: id={row_id}, 5 concurrent UPDATEs fired")

# Scenario 3: Ghost — INSERT then immediate DELETE before DMS flush
def scenario_ghost(events):
    conn = connect()
    conn.autocommit = True
    with conn.cursor() as cur:
        cur.execute(
            "INSERT INTO transactions (user_id, amount, status) VALUES (3, 0.01, 'pending') RETURNING id"
        )
        row_id = cur.fetchone()[0]
        log_event(events, "I", row_id, "ghost-insert")
        cur.execute("DELETE FROM transactions WHERE id = %s", (row_id,))
        log_event(events, "D", row_id, "ghost-delete")
    conn.close()
    print(f"Ghost: id={row_id} — inserted and immediately deleted")

# Scenario 4: Large transaction — 5000 rows in one commit
def scenario_large_txn(events):
    conn = connect()
    conn.autocommit = False
    with conn.cursor() as cur:
        cur.execute("""
            INSERT INTO transactions (user_id, amount, status)
            SELECT (generate_series % 2000) + 1, RANDOM() * 1000, 'pending'
            FROM generate_series(1, 5000)
            RETURNING id
        """)
        ids = [r[0] for r in cur.fetchall()]
        conn.commit()
    conn.close()
    for row_id in ids[:10]:
        log_event(events, "I", row_id, "large-txn")
    print(f"Large txn: 5000 rows in one commit. Sample ids: {ids[:5]}")

# Scenario 5: Rollback — changes DMS must NOT surface
def scenario_rollback(events):
    conn = connect()
    conn.autocommit = False
    with conn.cursor() as cur:
        cur.execute(
            "INSERT INTO transactions (user_id, amount, status) VALUES (5, 999.99, 'pending') RETURNING id"
        )
        row_id = cur.fetchone()[0]
        log_event(events, "I", row_id, "rollback-ROLLED-BACK")
        conn.rollback()
    conn.close()
    print(f"Rollback: id={row_id} inserted then rolled back — DMS must NOT surface this")

SCENARIOS = {
    "burst":      scenario_burst,
    "concurrent": scenario_concurrent,
    "ghost":      scenario_ghost,
    "large-txn":  scenario_large_txn,
    "rollback":   scenario_rollback,
}

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--scenario", required=True, choices=SCENARIOS.keys())
    parser.add_argument("--output",   default="stress_events.csv")
    args = parser.parse_args()

    events = []
    print(f"\nScenario : {args.scenario}")
    print(f"Started  : {now_utc()}\n")
    SCENARIOS[args.scenario](events)

    with open(args.output, "w", newline="") as f:
        writer = csv.DictWriter(f, fieldnames=["op", "id", "triggered_at", "note"])
        writer.writeheader()
        writer.writerows(events)
    print(f"\nLogged {len(events)} events → {args.output}")

if __name__ == "__main__":
    main()
```

## Task 8: Analysis

**Files:**
- Modify: `2-cdc-dms/analysis/analysis.py`

- [ ] **Step 1: Write analysis.py**

```python
"""
Post-run analysis of CDC pipeline metrics.

Reads from S3:
  - metrics/oltp/*.csv      — committed_at logs from OLTP workload
  - change-log/CDC/**/*.csv — DMS CDC change events

Outputs to --output-dir:
  - lag_table.txt           — p50/p95/p99 lag by op type
  - duplicates.txt          — duplicate event count and rate
  - event_trace.png         — timeline of change events

Usage:
    python analysis.py --bucket <bucket> --output-dir 2-cdc-dms/analysis/output/
"""
import argparse
import io
import os
from pathlib import Path
from collections import Counter

import boto3
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.dates as mdates

# ---------- S3 helpers ----------

def list_keys(s3, bucket, prefix):
    paginator = s3.get_paginator("list_objects_v2")
    keys = []
    for page in paginator.paginate(Bucket=bucket, Prefix=prefix):
        for obj in page.get("Contents", []):
            keys.append(obj["Key"])
    return sorted(keys)

def read_csv_s3(s3, bucket, key):
    obj = s3.get_object(Bucket=bucket, Key=key)
    return pd.read_csv(io.BytesIO(obj["Body"].read()), dtype=str)

def load_dataframes(s3, bucket, prefix):
    keys = list_keys(s3, bucket, prefix)
    if not keys:
        return pd.DataFrame()
    return pd.concat([read_csv_s3(s3, bucket, k) for k in keys], ignore_index=True)

# ---------- Lag ----------

def compute_lag(committed_df, cdc_df):
    c = committed_df.copy()
    c["id"]           = c["id"].astype(str)
    c["committed_at"] = pd.to_datetime(c["committed_at"], utc=True)

    e = cdc_df.copy()
    e["id"]          = e["id"].astype(str)
    e["_timestamp"]  = pd.to_datetime(e["_timestamp"], utc=True, errors="coerce")
    e = e.rename(columns={"Op": "op"})

    merged = pd.merge(c, e[["op", "id", "_timestamp"]], on=["op", "id"], how="inner")
    merged["lag_seconds"] = (merged["_timestamp"] - merged["committed_at"]).dt.total_seconds()
    return merged[merged["lag_seconds"] >= 0]

def write_lag_table(lag_df, out_dir):
    lines = ["Lag Analysis (seconds)\n", "=" * 40 + "\n"]
    for op, label in [("I", "INSERT"), ("U", "UPDATE"), ("D", "DELETE")]:
        s = lag_df[lag_df["op"] == op]["lag_seconds"]
        if s.empty:
            continue
        lines += [
            f"\nOp={op} ({label})  n={len(s)}\n",
            f"  p50 = {np.percentile(s, 50):.2f}s\n",
            f"  p95 = {np.percentile(s, 95):.2f}s\n",
            f"  p99 = {np.percentile(s, 99):.2f}s\n",
        ]
    text = "".join(lines)
    (out_dir / "lag_table.txt").write_text(text)
    print(text)

# ---------- Duplicates ----------

def write_duplicate_report(cdc_df, out_dir):
    if cdc_df.empty or "Op" not in cdc_df.columns:
        return
    total  = len(cdc_df)
    dupes  = cdc_df.duplicated(subset=["id", "_timestamp", "Op"], keep=False).sum()
    rate   = 100 * dupes / total if total else 0
    text   = (
        f"Duplicate Events\n{'='*40}\n"
        f"Total events : {total}\n"
        f"Duplicates   : {dupes}\n"
        f"Rate         : {rate:.2f}%\n"
    )
    (out_dir / "duplicates.txt").write_text(text)
    print(text)

# ---------- Event trace ----------

def write_event_trace(cdc_df, out_dir):
    if cdc_df.empty or "Op" not in cdc_df.columns:
        return
    df = cdc_df.copy()
    df["_timestamp"] = pd.to_datetime(df["_timestamp"], utc=True, errors="coerce")
    df["id_int"]     = pd.to_numeric(df["id"], errors="coerce")
    df = df.dropna(subset=["_timestamp", "id_int"])

    colors = {"I": "green", "U": "orange", "D": "red"}
    fig, ax = plt.subplots(figsize=(14, 6))
    for op, group in df.groupby("Op"):
        ax.scatter(group["_timestamp"], group["id_int"],
                   c=colors.get(op, "gray"), label=op, alpha=0.4, s=5)
    ax.xaxis.set_major_formatter(mdates.DateFormatter("%H:%M"))
    ax.set_xlabel("Wall clock time (UTC)")
    ax.set_ylabel("Transaction ID")
    ax.set_title("DMS CDC Event Trace")
    ax.legend(title="Op")
    plt.tight_layout()
    path = out_dir / "event_trace.png"
    plt.savefig(path, dpi=150)
    plt.close()
    print(f"Event trace saved to {path}")

# ---------- Main ----------

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--bucket",          required=True)
    parser.add_argument("--output-dir",      default="2-cdc-dms/analysis/output")
    parser.add_argument("--metrics-prefix",  default="metrics/oltp")
    parser.add_argument("--cdc-prefix",      default="change-log/CDC")
    args = parser.parse_args()

    out_dir = Path(args.output_dir)
    out_dir.mkdir(parents=True, exist_ok=True)

    s3 = boto3.client("s3")

    print("Loading committed_at logs...")
    committed_df = load_dataframes(s3, args.bucket, args.metrics_prefix)
    print(f"  {len(committed_df)} events")

    print("Loading CDC events...")
    cdc_df = load_dataframes(s3, args.bucket, args.cdc_prefix)
    print(f"  {len(cdc_df)} events")

    if not committed_df.empty and not cdc_df.empty:
        print("\nComputing lag...")
        lag_df = compute_lag(committed_df, cdc_df)
        write_lag_table(lag_df, out_dir)

    print("\nChecking duplicates...")
    write_duplicate_report(cdc_df, out_dir)

    print("\nGenerating event trace...")
    write_event_trace(cdc_df, out_dir)

    print(f"\nAll outputs in {out_dir}/")

if __name__ == "__main__":
    main()
```
